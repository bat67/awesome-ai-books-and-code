ld为样本例数

ld=400;

% 产生2*ld的矩阵
x=rand(2,ld);

% 将x转换到[-1.5 1.5]之间
x=(x-0.5)*1.5*2;

% x的第一行为x1，第二行为x2.
x1=x(1,:);
x2=x(2,:);

% 计算网络输出F值
F=20+x1.^2-10*cos(2*pi*x1)+x2.^2-10*cos(2*pi*x2);
